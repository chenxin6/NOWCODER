public class Solution {
    public int singleNumber(int[] A) {
    	int result = 0;
    	for (int i = 0; i < 32; i++) {
    		int sum = 0;
    		for (int j = 0; j < A.length; j++) {
    			sum += (A[j] >> i) & 1;
    		}
    		sum = sum % 2;
    		result |= sum << i;
    	}
        return result;
    }
}
